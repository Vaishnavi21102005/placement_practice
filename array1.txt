    {
        for(j=0;j<b;j++)
            scanf("%d",&arr[i][j]);
    }
    for(i=0;i<=a-2;i++)
    {
        for(j=0;j<=b-2;j++)
        {
            if(arr[i][j]!=arr[i+1][j+1])
                break;
        }
        if(j<=b-2)
         break;
    }

            printf("Not");
