#include<stdio.h>
int main()
{
    int size;
    scanf("%d",&size);
    char str[size];
    scanf("%s",str);
    int i,count=0,count1=0,count2=0,count3=0;
    for(i=0;str[i]!='\0';i++)
    {
        if(str[i]>='A' && str[i]<='Z')
        {
            count++;
        }
        else if(str[i]>='a' && str[i]<='z')
            count1++;
        else if(str[i]>='0' && str[i]<='9')
            count2++;
        else
            count3++;
    }
     printf("%d %d %d %d",count,count1,count2,count3);
}
